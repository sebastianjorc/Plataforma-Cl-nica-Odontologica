CREATE TABLE IF NOT EXISTS `Historial de servicios` (
	`ID_PRACTICANTE` VARCHAR(45) NOT NULL,
	`MES` VARCHAR(45) NULL,
	`NUMERO DÍA` VARCHAR(45) NOT NULL,
	`ID SERVICIO` VARCHAR(45) NULL,
	`CALIFICACION` VARCHAR(45) NULL,
	PRIMARY KEY (`ID_PRACTICANTE`)
)
