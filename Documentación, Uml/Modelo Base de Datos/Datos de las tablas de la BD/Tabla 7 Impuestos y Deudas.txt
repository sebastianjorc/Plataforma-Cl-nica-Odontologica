CREATE TABLE IF NOT EXISTS `Egresos Mensuales` (
	`ANO` VARCHAR(45) NOT NULL,
	`MES` VARCHAR(45) NOT NULL,
	`INVERSION INFRAESTRUCTURA` INT NOT NULL,
	`INVERSION TECNOLOGICA` INT NULL,
	`INVERSION EN EQUIPO MEDICO` INT NULL,
	PRIMARY KEY (`ANO`,`MES`)
)
